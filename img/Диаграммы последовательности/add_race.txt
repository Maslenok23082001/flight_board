@startuml
actor "Администратор, диспетчер" as pols

entity "База данных рейсов" as bd_races
entity "Проверка корректности данных" as korrect


boundary "Меню администратора" as menu_admin
boundary "Окно управления рейсами" as control_races
boundary "Окно добавления рейса" as add_race

 
 
pols -> menu_admin : Переход в меню администратора
menu_admin -> control_races : Переход в окно управления рейсами
activate control_races
control_races -> add_race : Ввод данных для добавления рейса
activate add_race
add_race -> korrect : Проверка введённых данных
activate korrect
korrect -> add_race : Некорректные данные
korrect -> bd_races : Успешное добавление рейса
@enduml
