boundary "Главное окно" as main_window

vhod -> bd_users : Проверка должности пользователя
bd_users -> vhod : Параметры не совпадают
vhod -> bd_users : Проверка наличия пользователя
bd_users -> vhod : Пользователь найден
vhod -> main_window : Доступ к главному окно
