@startuml
actor Пассажир

entity "База данных пользователей" as bd_users

boundary "Окно авторизации" as vhod
boundary "Главное окно" as main_window
 
Пассажир -> vhod : Ввод данных пользователя(логин, пароль)
Пассажир -> vhod : Сигнал о завершении ввода

vhod -> bd_users : Проверка на наличие пользователя в базе данных
vhod -> bd_users : Проверка должности пользователя
bd_users -> vhod : Параметры не совпадают
vhod -> bd_users : Проверка наличия пользователя
bd_users -> vhod : Пользователь найден
vhod -> main_window : Доступ к главному окно
@enduml
